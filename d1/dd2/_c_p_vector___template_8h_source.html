985 void _CustomSort_Helper(const Sorting::Callback<T> CompareFunction, unsigned int lower, unsigned int upper)
986 {
987 // Limits check
988 if(lower > upper)
989 {
990 unsigned int tmp = lower;
991 lower = upper;
992 upper = tmp;
993 }
994
995 // Calculate new Pivot

998

1001 if(pivot < upper-1)_CustomSort_Helper(CompareFunction,pivot+1,upper);

1004 unsigned int _CustomSort_Partition(int8_t (*CompareFunction)(const T& a, const T& b), unsigned int lower, unsigned int upper, unsigned int pivot)
1005 {
1006 // Store Pivot value at the Highest index (upper limit)
1007 swap(pivot,upper);
1008
1009 // Initialize the exchange index at the lower limit
1010 int index = lower;
1011
1012 // Cycle through each element
1013 for(unsigned int i = lower ; i < upper; i++)
1014 {
1015 // Comparing the n'th element against the pivot
